# PAW CRUD

## Tabel Database
CREATE TABLE POST (
id INT(10) UNSIGNED NOT NULL AUTO_INCREMENT,  
title VARCHAR(100) NOT NULL,  
content MEDIUMTEXT NOT NULL,  
PRIMARY KEY (id)  
);
